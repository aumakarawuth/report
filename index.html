function loadData() { document.getElementById('topbar-label').textContent = 'กำลังโหลด...'; // โหลด archive list fetch(GAS_URL + '?action=getArchivedMonths', { cache: 'no-cache' }) .then(r => r.json()) .then(archives => renderArchiveSelector(archives || [])) .catch(() => {}); var targetSheet = archiveSheetName || 'รายการ'; // ── โหลด records และ members แยกกัน ไม่ chain ── Promise.all([ fetch(GAS_URL + '?action=getDashboardDataFromSheet&sheetName=' + encodeURIComponent(targetSheet), { cache: 'no-cache' }) .then(r => r.json()) .catch(() => ({ records: [] })), fetch(GAS_URL + '?action=getMembersData', { cache: 'no-cache' }) .then(r => r.json()) .catch(() => ({ members: [] })) ]) .then(function(results) { var recResult = results[0]; var memResult = results[1]; if (recResult && recResult.error) { showError(recResult.error); return; } processRecords((recResult && recResult.records) || []); if (memResult && memResult.members) { processMemberData(memResult.members); } finishLoad(); }) .catch(function(err) { showError('โหลดไม่สำเร็จ: ' + (err.message || String(err))); }); }
